import java.util.Scanner;
class EcuacionLineal {
    private double a;
    private double b;
    private double c;
    private double d;
    private double e;
    private double f;
    public EcuacionLineal(double a, double b, double double d, double e, double f) {
        this.a = a;
        this.b = b;
        this.c = c;
        this.d = d;
        this.e = e;
        this.f = f;
    private double determinante() {
        return a * d - b * c;
    }
    public boolean tieneSolucion() {
        return determinante() != 0;
    }
    public double getX() {
        return (e * d - b * f) / determinante();
    }
    public double getY() {
        return (a * f - e * c) / determinante();
    }
}
public class Main {
    public static void main(String[] args) {
        Scanner entrada = new Scanner(System.in);
        System.out.println("Ingrese a, b, c, d, e, f:");
        double a = entrada.nextDouble();
        double b = entrada.nextDouble();
        double c = entrada.nextDouble();
        double d = entrada.nextDouble();
        double e = entrada.nextDouble();
        double f = entrada.nextDouble();
        EcuacionLineal ecuacion = new EcuacionLineal(a, b, c, d, e, f);
        if (ecuacion.tieneSolucion()) {
            System.out.println("x = " + ecuacion.getX()+ ", y = " + ecuacion.getY());
        } else {
            System.out.println("La ecuacion no tiene solucion");
        }
        entrada.close();
    }
}
